cmake_minimum_required(VERSION 3.30)
project(untitled LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

add_executable(untitled main.cpp)

# Local raylib distribution (header + static library)
set(RAYLIB_ROOT "${CMAKE_SOURCE_DIR}/raylib-5.5_win64_mingw-w64")
set(RAYLIB_INCLUDE_DIR "${RAYLIB_ROOT}/include")
set(RAYLIB_LIB "${RAYLIB_ROOT}/lib/libraylib.a")

if (MSVC)
    message(FATAL_ERROR "This project is configured to link MinGW raylib (.a). Select a MinGW/GCC toolchain in CLion.")
endif()
if (NOT EXISTS "${RAYLIB_INCLUDE_DIR}/raylib.h")
    message(FATAL_ERROR "raylib.h not found at: ${RAYLIB_INCLUDE_DIR}/raylib.h")
endif()
if (NOT EXISTS "${RAYLIB_LIB}")
    message(FATAL_ERROR "libraylib.a not found at: ${RAYLIB_LIB}")
endif()

add_library(raylib STATIC IMPORTED GLOBAL)
set_target_properties(raylib PROPERTIES
        IMPORTED_LOCATION "${RAYLIB_LIB}"
        INTERFACE_INCLUDE_DIRECTORIES "${RAYLIB_INCLUDE_DIR}"
)

target_link_libraries(untitled PRIVATE raylib)

# raylib's platform libs (common for MinGW on Windows)
if (WIN32)
    target_link_libraries(untitled PRIVATE winmm gdi32 opengl32)
endif()
